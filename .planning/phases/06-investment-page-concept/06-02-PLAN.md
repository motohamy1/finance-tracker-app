**Interface addition:**
```typescript
getSummaryByTicker: (ticker: string) => {
  totalInvestedCents: number;       // Sum of (shares × price) + fees for all buy trades of this ticker
  totalRealizedCents: number;       // Sum of realizedPnlCents from PnL pairs for this ticker
  netPnlCents: number;              // totalRealizedCents - totalFees on matched pairs
  pnlMultiplier: number | null;     // (totalInvested + netPnl) / totalInvested, null if no buys
  pnlPercent: number | null;        // ((totalInvested + netPnl) / totalInvested - 1) × 100
  buyCount: number;                 // Number of buy trades for this ticker
  sellCount: number;                // Number of sell trades for this ticker
  pairCount: number;                // Number of completed PnL pairs
  totalBuyVolumeCents: number;      // Total capital deployed (shares × price, no fees)
  totalSellVolumeCents: number;     // Total capital returned (shares × price, no fees)
};
```

**Implementation (add to store creator, after getCategorySummary):**

```typescript
getSummaryByTicker: (ticker) => {
  const upperTicker = ticker.toUpperCase();
  const tickerTrades = get().trades.filter(t => t.ticker === upperTicker);
  const pairs = calculatePnLPairs(tickerTrades);
  
  const buyTrades = tickerTrades.filter(t => t.direction === 'buy');
  const sellTrades = tickerTrades.filter(t => t.direction === 'sell');
  
  const totalInvestedCents = buyTrades.reduce(
    (sum, t) => sum + t.shares * t.pricePerShareCents + (t.feesCents || 0), 0
  );
  const totalBuyVolumeCents = buyTrades.reduce(
    (sum, t) => sum + t.shares * t.pricePerShareCents, 0
  );
  const totalSellVolumeCents = sellTrades.reduce(
    (sum, t) => sum + t.shares * t.pricePerShareCents, 0
  );
  const totalRealizedCents = pairs.reduce((sum, p) => sum + p.realizedPnlCents, 0);
  const totalFeesCents = pairs.reduce((sum, p) => sum + p.buyFeesCents + p.sellFeesCents, 0);
  const netPnlCents = totalRealizedCents - totalFeesCents;
  
  const pnlMultiplier = totalInvestedCents > 0
    ? (totalInvestedCents + netPnlCents) / totalInvestedCents
    : null;
  const pnlPercent = pnlMultiplier !== null
    ? (pnlMultiplier - 1) * 100
    : null;
  
  return {
    totalInvestedCents,
    totalRealizedCents,
    netPnlCents,
    pnlMultiplier,
    pnlPercent,
    buyCount: buyTrades.length,
    sellCount: sellTrades.length,
    pairCount: pairs.length,
    totalBuyVolumeCents,
    totalSellVolumeCents,
  };
},
```
