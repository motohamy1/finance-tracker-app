```typescript
export interface FieldAccuracy {
  field: string;
  correct: number;
  total: number;
  accuracy: number;  // 0.0–1.0
}

export interface AccuracyReport {
  overallAccuracy: number;       // 0.0–1.0
  fieldAccuracies: FieldAccuracy[];  // Per-field breakdown
  totalFields: number;
  correctFields: number;
  platform: Platform;
  extractionMethod: 'template' | 'regex';
  sampleCount: number;           // Always 1 for single evaluation; 5+ for batch
}

export interface ComparisonReport {
  regexAccuracy: number;
  aiAccuracy: number;
  improvement: number;           // Positive = AI better, negative = regex better
  fieldsImproved: string[];      // Which fields AI extracted better
  fieldsRegressed: string[];     // Which fields regex extracted better (should be empty)
  verdict: 'ai_wins' | 'regex_wins' | 'tie';
  regexReport: AccuracyReport;
  aiReport: AccuracyReport;
}
```

**Functions to implement:**

1. **`evaluateAccuracy(groundTruth: OCRResult, extracted: OCRResult): AccuracyReport`**
   - Compares 5 fields: ticker, shares, pricePerShare, tradeDate, direction
   - For each field: if groundTruth[field] === extracted[field] → correct; else → incorrect
   - Handle nulls: if both are null, count as correct (null-safe comparison)
   - Handle numbers: shares and pricePerShare compared with exact equality (no tolerance)
   - Handle strings: case-insensitive comparison for ticker and direction
   - Handle dates: string comparison (both ISO format YYYY-MM-DD)
   - overallAccuracy = correctFields / totalFields
   - Populates platform and extractionMethod from extracted.aiMeta (falls back to 'generic'/'regex')

2. **`compareExtractions(regexResult: OCRResult, aiResult: OCRResult, groundTruth: OCRResult): ComparisonReport`**
   - Runs evaluateAccuracy for both regexResult and aiResult against groundTruth
   - Computes improvement = aiAccuracy - regexAccuracy
   - Lists fields where AI improved (ai correct, regex wrong) and regressed (ai wrong, regex correct)
   - Sets verdict based on improvement sign

3. **`formatAccuracyReport(report: AccuracyReport): string`**
   - Returns formatted string: `"AI (robinhood/template): 80.0% overall accuracy (4/5 fields). Ticker: 100%, Shares: 100%, Price: 100%, Direction: 100%, Date: 0%"`
   - Platform and extraction method from report

4. **`formatComparisonReport(report: ComparisonReport): string`**
   - Returns formatted string: `"AI wins: 80.0% vs regex 40.0% (+40.0% improvement). Fields improved: ticker, shares. Fields regressed: none."`

```typescript
describe('OCR Accuracy Regression Suite', () => {
  it('achieves ≥80% overall accuracy across all 15 platform-specific screenshots', () => {
    let totalCorrect = 0;
    let totalFields = 0;
    const perSampleResults: { description: string; accuracy: number; platform: string }[] = [];

    for (const tc of testCases) {
      const result = parseTradeFromText(tc.rawText);
      const report = evaluateAccuracy(tc.groundTruth, result);
      totalCorrect += report.correctFields;
      totalFields += report.totalFields;
      perSampleResults.push({
        description: tc.description,
        accuracy: report.overallAccuracy,
        platform: report.platform,
      });
    }

    const overallAccuracy = totalCorrect / totalFields;
    expect(overallAccuracy).toBeGreaterThanOrEqual(0.80);

    // Log per-sample results for debugging
    for (const r of perSampleResults) {
      console.log(`  ${r.platform}: ${r.description} — ${(r.accuracy * 100).toFixed(0)}%`);
    }
    console.log(`  OVERALL: ${(overallAccuracy * 100).toFixed(1)}% (${totalCorrect}/${totalFields} fields)`);
  });
});
```

```typescript
/**
 * Check if a field was missing from OCR extraction.
 * When AI extraction metadata is available, uses per-field confidence
 * to determine if the field was confidently extracted.
 * Falls back to null-check for Phase 2 regex-only results.
 */
export function isMissingFromOCR(ocrResult: OCRResult | null, fieldKey: string): boolean {
  if (!ocrResult) return true;

  // If AI metadata exists, use per-field confidence
  if (ocrResult.aiMeta?.perFieldConfidence) {
    const confidence = ocrResult.aiMeta.perFieldConfidence[fieldKey];
    // Field is "missing" if confidence is 0 or not present
    return confidence === undefined || confidence === 0;
  }

  // Legacy: null-check for regex-only extraction
  switch (fieldKey) {
    case 'ticker': return ocrResult.ticker === null;
    case 'shares': return ocrResult.shares === null;
    case 'pricePerShare':
    case 'pricePerShareCents':
      return ocrResult.pricePerShare === null;
    case 'tradeDate': return ocrResult.tradeDate === null;
    case 'direction': return ocrResult.direction === null;
    default: return true;
  }
}
```

```tsx
        name={ocrResult.confidence >= 0.75 ? 'checkmark-circle' : 'warning'}
        size={16}
        color={ocrResult.confidence >= 0.75 ? '#059669' : '#D97706'}
      />
      <Text style={styles.confidenceText}>
        Extraction Confidence: {Math.round(ocrResult.confidence * 100)}%
      </Text>
    </View>

    {/* AI metadata row — shows platform and extraction method */}
    {ocrResult.aiMeta && ocrResult.aiMeta.platform !== 'generic' && (
      <View style={styles.aiMetaRow}>
        <Ionicons name="hardware-chip-outline" size={14} color="#6366F1" />
        <Text style={styles.aiMetaText}>
          AI-enhanced • {ocrResult.aiMeta.platform.charAt(0).toUpperCase() + ocrResult.aiMeta.platform.slice(1)} template
        </Text>
        <View style={[
          styles.aiPill,
          ocrResult.aiMeta.platformConfidence >= 0.7 ? styles.aiPillHigh : styles.aiPillLow,
        ]}>
          <Text style={styles.aiPillText}>
            {Math.round(ocrResult.aiMeta.platformConfidence * 100)}% match
          </Text>
        </View>
      </View>
    )}

    {/* Generic regex fallback indicator */}
    {(!ocrResult.aiMeta || ocrResult.aiMeta.platform === 'generic') && (
      <View style={styles.aiMetaRow}>
        <Ionicons name="text-outline" size={14} color="#94A3B8" />
        <Text style={styles.aiMetaTextMuted}>Generic text extraction</Text>
      </View>
    )}
  </View>
)}
```
