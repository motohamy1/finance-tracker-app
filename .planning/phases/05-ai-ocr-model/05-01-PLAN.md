  extract: (text: string) => Partial<Pick<OCRResult, 'ticker' | 'shares' | 'pricePerShare' | 'tradeDate' | 'direction'>>;
}

export const EXTRACTION_TEMPLATES: ExtractionTemplate[] = [];

{
  platform: 'robinhood',
  extract: (text: string) => {
    const result: Partial<Pick<OCRResult, 'ticker' | 'shares' | 'pricePerShare' | 'tradeDate' | 'direction'>> = {};
    
    // Direction: "Market Buy" or "Market Sell" or "Order Buy" or "Order Sell"
    if (/Market\s+Buy|Order\s+Buy/i.test(text)) result.direction = 'buy';
    else if (/Market\s+Sell|Order\s+Sell/i.test(text)) result.direction = 'sell';

    const sharesMatch = text.match(/(?:shares?|sh|quantity)\s*[:]?\s*(\d+(?:\.\d+)?)/i)

    if (sharesMatch) result.shares = parseInt(sharesMatch[1], 10) || null;
    
    // Price: "@ $N" or "at $N" or "Limit Price $N" or "Price $N"
    const priceMatch = text.match(/(?:@|at|price)\s*\$?\s*(\d+(?:\.\d{1,2})?)/i);
    if (priceMatch) result.pricePerShare = parseFloat(priceMatch[1]) || null;
    
    // Ticker: uppercase after "shares of" or "sh of" or pattern "N TICKER"
    const tickerMatch = text.match(/(?:shares?\s*(?:of|in)?|sh\s*(?:of|in)?)\s+([A-Z]{1,5})/i)

      const noise = ['BUY', 'SELL', 'LMT', 'MKT', 'STP', 'GTC', 'DAY', 'IOC', 'RH', 'USD', 'THE'];
      if (!noise.includes(ticker)) result.ticker = ticker;
    }
    
    return result;

**Step 6: Create `extractWithTemplate` function (new export)**
```typescript
/**
 * Extract trade fields using platform-specific template, with field-level fallback.
 * Returns extracted fields + per-field confidence so caller can merge with generic extraction.
 */
export function extractWithTemplate(
  rawText: string,
  platform: Platform
): { fields: Partial<OCRResult>; perFieldConfidence: Record<string, number> } {
  const template = EXTRACTION_TEMPLATES.find(t => t.platform === platform);
  const perFieldConfidence: Record<string, number> = {};
  
  if (!template) {
    // No template for this platform — return empty, all confidence 0
    for (const field of ['ticker', 'shares', 'pricePerShare', 'tradeDate', 'direction']) {
      perFieldConfidence[field] = 0;
    }
    return { fields: {}, perFieldConfidence };
  }
  
  const fields = template.extract(rawText);
  
  // Per-field confidence: 1.0 if extracted, 0 if null
  for (const field of ['ticker', 'shares', 'pricePerShare', 'tradeDate', 'direction']) {
    const key = field as keyof typeof fields;
    perFieldConfidence[field] = fields[key] != null ? 1.0 : 0;
  }
  
  return { fields, perFieldConfidence };
}
```
